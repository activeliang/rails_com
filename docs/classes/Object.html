def for_each_process(&block)
  fetch(:sidekiq_processes).times do |idx|
    if idx == 0
      pid_file = fetch :sidekiq_pid
    else
      pid_file = "#{fetch :sidekiq_pid}-#{idx}"
    end

    yield(pid_file, idx)
  end
end
